\begin{Exercise}
	\begin{proof}
		By hypothesis, we know $f(0) = a_0$ trivially. Notice that $f(t) = \det(A-\lambda I)$ and hence $f(0) = \det(A - 0 I) = \det(A)$. They follow that $f(0) = a_0 = \det(A)$. Moreover, 
		$$
		A\text{ is invertible}
		\iff \det(A)\neq 0
		\iff a_0\neq 0.
		$$
	\end{proof}
\end{Exercise}
