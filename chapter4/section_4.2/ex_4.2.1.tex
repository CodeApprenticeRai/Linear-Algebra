		\item[(e)]
		\begin{answer}
			False.
		\end{answer}
		\begin{solution}
			$\det\begin{pmatrix}
			2 & 0 \\
			0 & 1
			\end{pmatrix} = 2$ and $\det(I) = 1$; however, they do not equal.
		\end{solution}
		
		\item[(f)]
		\begin{answer}
			False.
		\end{answer}
		\begin{solution}
			$\det\begin{pmatrix}
			1 & k \\
			0 & 1
			\end{pmatrix} = 1$ and $k\det(I) = k$; however, $\det\begin{pmatrix}
			1 & k \\
			0 & 1
			\end{pmatrix} \neq k\det(I)$.
		\end{solution}
