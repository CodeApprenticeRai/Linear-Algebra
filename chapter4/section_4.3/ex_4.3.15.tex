\begin{Exercise}
	\begin{proof}
		Since $A$ is similar to $B$, then there exists $Q\in M_{n\times n}(F)$ such that $A = Q^{-1} B Q$. It follows that 
		\begin{gather*}
		\det(A) = \det(Q^{-1} B Q) = \det(Q^{-1})\det(B)\det(Q) = (\det(Q^{-1})\det(Q))\det(B) \\
		= \det(Q^{-1}Q)\det(B) = \det(I)\det(B) = \det(I B) = \det(B).
		\end{gather*}
	\end{proof}
\end{Exercise}
