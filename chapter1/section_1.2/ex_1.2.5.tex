\begin{Exercise}
	\begin{solution}
		The upstream crossing matrix is
		$
		\begin{pmatrix}
		8 & 3 & 1 \\
		3 & 0 & 0 \\
		3 & 0 & 0
		\end{pmatrix}.
		$
		
		The downstream crossing matrix is
		$
		\begin{pmatrix}
		9 & 1 & 4 \\
		3 & 0 & 0 \\
		1 & 1 & 0
		\end{pmatrix}.
		$
		
		The total of crossing matrix is
		$
		\begin{pmatrix}
		8 & 3 & 1 \\
		3 & 0 & 0 \\
		3 & 0 & 0
		\end{pmatrix}
		+ \begin{pmatrix}
		9 & 1 & 4 \\
		3 & 0 & 0 \\
		1 & 1 & 0
		\end{pmatrix}
		= \begin{pmatrix}
		17 & 4 & 5 \\
		6 & 0 & 0 \\
		4 & 1 & 0
		\end{pmatrix}.
		$
	\end{solution}
\end{Exercise}
