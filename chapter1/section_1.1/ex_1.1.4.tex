\begin{Exercise}
\begin{answer}
$(0, 0, \cdots, 0)\in\mathbb{R}^n$.
\end{answer}
\begin{proof}
Let $x := (a_1, a_2, \cdots, a_n)\in\mathbb{R}^n$, and $\mathbf{0} := (0, 0, \cdots, 0)\in\mathbb{R}^n$, then 
$$
x+\mathbf{0}
= (a_1 + 0, a_2 + 0, \cdots, a_n + 0)
= (a_1, a_2, \cdots, a_n)
= x.
$$
Hence $\mathbf{0}$ is the zero vector.
\end{proof}
\end{Exercise}
