\begin{answer}
$\left\{\begin{pmatrix}
5 \\
0
\end{pmatrix} + t\begin{pmatrix}
-3 \\
1
\end{pmatrix} : t\in \mathbb{R} \right\}$.
\end{answer}
\begin{solution}
Since $\begin{cases}
x_1 = 5 \\
x_2 = 0
\end{cases}$ is a particular solution. Then the answer follows from Exercise 3.3.2(a).
\end{solution}

\begin{answer}

\end{pmatrix} + t\begin{pmatrix}

\begin{solution}

x_1 = 1 \\
