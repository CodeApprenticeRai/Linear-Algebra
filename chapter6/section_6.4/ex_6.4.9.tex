\begin{Exercise}
	\begin{proof}
		Since $T$ is normal, if $\|T x\| = 0$ for arbitrary $x\in V$, then $\|T^* x \| = 0$. This converse also holds. Hence $N(T) = N(T^*)$. By Exercise 6.3.12, we know 
		$$
		R(T^*) = N(T)^{\perp} = N(T^*)^{\perp} = \big(R(T)^{\perp}\big)^{\perp} = R(T).
		$$
	\end{proof}
\end{Exercise}
