\begin{Exercise}
	\begin{answer}
		$T(8,11) = (5,-3,16)$.
	\end{answer}
	\begin{proof}
		Since $\{(1,1),(2,3)\}$ is a basis for $\mathbb{R}^2$, by Theorem 2.6, we know the linear transformation $T$ exists.
		Moreover, $T(8,11) = 2T(1,1)+3T(2,3) = 2(1,0,2)+3(1,-1,4) = (5,-3,16)$.
	\end{proof}
\end{Exercise}
